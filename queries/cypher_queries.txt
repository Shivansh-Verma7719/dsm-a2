// Q1. Top 10 users by direct friend count; include total reviews and mean star rating given.

OPTIONAL MATCH (u)-[:FRIENDS_WITH]-(f:User)
WITH u, count(DISTINCT f) AS friend_count

  u.name AS user_name,
  friend_count,
  total_review_count,
  round(coalesce(mean_star_rating_given, 0.0) * 100.0) / 100.0 AS mean_star_rating_given

// Q2. Top 3 businesses per state (>=50 reviews)

// Q2. For each state, top 3 businesses by average review stars among businesses with >= 50 reviews.
MATCH (b:Business)<-[:REVIEWS]-(r:Review)
WHERE b.state IS NOT NULL AND trim(b.state) <> ""
WITH b, count(r) AS review_count, avg(r.stars) AS avg_star_rating
WHERE review_count >= 50
OPTIONAL MATCH (b)-[:IN_CATEGORY]->(c:Category)
WITH b, review_count, avg_star_rating, collect(DISTINCT c.name) AS categories
ORDER BY b.state, avg_star_rating DESC, review_count DESC, b.name
WITH b.state AS state,
     collect({
       business_id: b.id,
       business_name: b.name,
       city: b.city,
       categories: categories,
       avg_star_rating: round(avg_star_rating * 100.0) / 100.0,
       review_count: review_count
     }) AS ranked
UNWIND range(0, size(ranked) - 1) AS i
WITH state, ranked[i] AS row, i + 1 AS rank
WHERE rank <= 3
RETURN
  state,
  rank,
  row.business_id AS business_id,
  row.business_name AS business_name,
  row.city AS city,
  coalesce(row.categories[0], "Uncategorized") AS category,
  row.avg_star_rating AS avg_star_rating,
  row.review_count AS review_count
ORDER BY state, rank;

// Q3. Top 10 users by number of distinct reviewed cities; include mean star rating per city.

MATCH (u:User)-[:WROTE]->(r:Review)-[:REVIEWS]->(b:Business)
WHERE b.city IS NOT NULL AND trim(b.city) <> ""
WITH u, b.city AS city, avg(r.stars) AS mean_star_rating_city, count(r) AS reviews_in_city
WITH u,
     collect({
       city: city,
       mean_star_rating: round(mean_star_rating_city * 100.0) / 100.0,
       reviews: reviews_in_city
     }) AS city_stats,
     count(city) AS distinct_cities
RETURN
  u.id AS user_id,
  u.name AS user_name,
  distinct_cities,
  city_stats
ORDER BY distinct_cities DESC, user_id
LIMIT 10;

// Q4. Category-specialized users vs overall rating

// Q4. Users with >=5 reviewed businesses in a chosen category; compare category mean vs overall mean stars.

WITH $category AS target_category
MATCH (u:User)-[:WROTE]->(r_cat:Review)-[:REVIEWS]->(b_cat:Business)-[:IN_CATEGORY]->(:Category {name: target_category})
WITH u, target_category, count(DISTINCT b_cat) AS distinct_businesses_in_category, avg(r_cat.stars) AS category_mean_star
WHERE distinct_businesses_in_category >= 5
MATCH (u)-[:WROTE]->(r_all:Review)
WITH u, target_category, distinct_businesses_in_category, category_mean_star,
     count(r_all) AS total_reviews_all,
     avg(r_all.stars) AS overall_mean_star
RETURN
  target_category AS category,
  u.id AS user_id,
  u.name AS user_name,
  distinct_businesses_in_category,
  total_reviews_all,
  round(category_mean_star * 100.0) / 100.0 AS category_mean_star,
  round(overall_mean_star * 100.0) / 100.0 AS overall_mean_star,
  round((category_mean_star - overall_mean_star) * 100.0) / 100.0 AS category_minus_overall
ORDER BY distinct_businesses_in_category DESC, total_reviews_all DESC
LIMIT 100;

// Q5. Cypher reproduction of Mongo volume-vs-rating

// Q5. Reproduce Mongo Section-5 Query 3 (review-volume vs average rating) in Cypher.

MATCH (b:Business)<-[:REVIEWS]-(r:Review)
WITH b, count(r) AS review_count, avg(r.stars) AS avg_star
WITH
  CASE
    WHEN review_count < 10 THEN '0-9'
    WHEN review_count < 50 THEN '10-49'
    WHEN review_count < 200 THEN '50-199'
    WHEN review_count < 500 THEN '200-499'
    WHEN review_count < 1000 THEN '500-999'
    WHEN review_count < 5000 THEN '1000-4999'
    ELSE '5000+'
  END AS volume_bucket,
  avg_star
RETURN
  volume_bucket,
  count(*) AS business_count,
  round(avg(avg_star) * 100.0) / 100.0 AS mean_business_rating
ORDER BY
  CASE volume_bucket
    WHEN '0-9' THEN 1
    WHEN '10-49' THEN 2
    WHEN '50-199' THEN 3
    WHEN '200-499' THEN 4
    WHEN '500-999' THEN 5
    WHEN '1000-4999' THEN 6
    ELSE 7
  END;
